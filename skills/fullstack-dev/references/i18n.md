# Internationalization (i18n) & Localization (l10n)

## Frontend i18n Setup

### Next.js Internationalization

```typescript
// next.config.js
module.exports = {
  i18n: {
    locales: ['en', 'zh', 'ja', 'ko', 'es', 'fr', 'de'],
    defaultLocale: 'en',
    localeDetection: true,
  },
};

// middleware.ts (for custom locale detection)
import { NextRequest, NextResponse } from 'next/server';

const LOCALES = ['en', 'zh', 'ja', 'ko'];
const DEFAULT_LOCALE = 'en';

export function middleware(request: NextRequest) {
  const pathname = request.nextUrl.pathname;
  
  // Check if pathname already has locale
  const pathnameHasLocale = LOCALES.some(
    locale => pathname.startsWith(`/${locale}/`) || pathname === `/${locale}`
  );
  
  if (pathnameHasLocale) return;
  
  // Detect locale from Accept-Language header
  const acceptLanguage = request.headers.get('accept-language') || '';
  const preferredLocale = LOCALES.find(locale => 
    acceptLanguage.includes(locale)
  ) || DEFAULT_LOCALE;
  
  // Redirect to localized path
  return NextResponse.redirect(
    new URL(`/${preferredLocale}${pathname}`, request.url)
  );
}

export const config = {
  matcher: ['/((?!api|_next/static|_next/image|favicon.ico).*)'],
};
```

### Locale Switcher

```tsx
// frontend/src/components/LocaleSwitcher.tsx
'use client';

import { useRouter, usePathname } from 'next/navigation';
import { useLocale } from 'next-intl';

const LOCALES = [
  { code: 'en', name: 'English', flag: '🇺🇸' },
  { code: 'zh', name: '中文', flag: '🇨🇳' },
  { code: 'ja', name: '日本語', flag: '🇯🇵' },
  { code: 'ko', name: '한국어', flag: '🇰🇷' },
];

export function LocaleSwitcher() {
  const router = useRouter();
  const pathname = usePathname();
  const currentLocale = useLocale();
  
  const switchLocale = (newLocale: string) => {
    const newPath = pathname.replace(`/${currentLocale}`, `/${newLocale}`);
    router.push(newPath);
  };
  
  return (
    <Select value={currentLocale} onValueChange={switchLocale}>
      <SelectTrigger>
        <SelectValue />
      </SelectTrigger>
      <SelectContent>
        {LOCALES.map(locale => (
          <SelectItem key={locale.code} value={locale.code}>
            {locale.flag} {locale.name}
          </SelectItem>
        ))}
      </SelectContent>
    </Select>
  );
}
```

## Date & Time Localization

### Timezone Handling

```typescript
// frontend/src/lib/datetime.ts
import { format, formatDistanceToNow, parseISO } from 'date-fns';
import { zhCN, ja, ko, enUS } from 'date-fns/locale';

const LOCALE_MAP = {
  en: enUS,
  zh: zhCN,
  ja: ja,
  ko: ko,
};

export function formatDate(
  date: string | Date,
  locale: string = 'en',
  formatStr: string = 'PPP'
): string {
  const d = typeof date === 'string' ? parseISO(date) : date;
  return format(d, formatStr, { locale: LOCALE_MAP[locale] || enUS });
}

export function formatRelativeTime(
  date: string | Date,
  locale: string = 'en'
): string {
  const d = typeof date === 'string' ? parseISO(date) : date;
  return formatDistanceToNow(d, { 
    addSuffix: true, 
    locale: LOCALE_MAP[locale] || enUS 
  });
}

// Usage
formatDate('2024-01-15', 'en');        // "Jan 15, 2024"
formatDate('2024-01-15', 'zh');        // "2024年1月15日"
formatRelativeTime('2024-01-10', 'en'); // "5 days ago"
formatRelativeTime('2024-01-10', 'zh'); // "5 天前"
```

```tsx
import { useTimezone } from '@/hooks/useTimezone';

export function DateTimeDisplay({ utcDate }: { utcDate: string }) {
  const { timezone, formatInTimezone } = useTimezone();
  
  return (
    <div>
      <span>{formatInTimezone(utcDate, 'PPPp')}</span>
      <span className="text-muted">{timezone}</span>
    </div>
  );
}

// Hook for user timezone
function useTimezone() {
  const [timezone, setTimezone] = useState(
    Intl.DateTimeFormat().resolvedOptions().timeZone
  );
  
  const formatInTimezone = (utcDate: string, formatStr: string) => {
    const date = new Date(utcDate);
    return formatInTimeZone(date, timezone, formatStr);
  };
  
  return { timezone, setTimezone, formatInTimezone };
}
```

## Number & Currency Formatting

```typescript
// frontend/src/lib/format.ts

// Number formatting
export function formatNumber(
  value: number,
  locale: string = 'en',
  options?: Intl.NumberFormatOptions
): string {
  return new Intl.NumberFormat(locale, options).format(value);
}

// Currency formatting
export function formatCurrency(
  value: number,
  currency: string = 'USD',
  locale: string = 'en'
): string {
  return new Intl.NumberFormat(locale, {
    style: 'currency',
    currency,
  }).format(value);
}

// Compact number (1K, 1M, etc.)
export function formatCompactNumber(
  value: number,
  locale: string = 'en'
): string {
  return new Intl.NumberFormat(locale, {
    notation: 'compact',
    compactDisplay: 'short',
  }).format(value);
}

// Usage
formatNumber(1234567.89, 'en');           // "1,234,567.89"
formatNumber(1234567.89, 'zh');           // "1,234,567.89"
formatCurrency(99.99, 'USD', 'en');       // "$99.99"
formatCurrency(99.99, 'CNY', 'zh');       // "¥99.99"
formatCurrency(99.99, 'JPY', 'ja');       // "¥100"
formatCompactNumber(1234567, 'en');       // "1.2M"
formatCompactNumber(1234567, 'zh');       // "123万"
```

## Backend i18n

### API Localization

```python
# backend/app/core/i18n.py
from typing import Dict, Any
import json
from pathlib import Path

class I18nService:
    def __init__(self, translations_dir: str = "translations"):
        self.translations: Dict[str, Dict[str, Any]] = {}
        self._load_translations(translations_dir)
    
    def _load_translations(self, directory: str):
        for file in Path(directory).glob("*.json"):
            locale = file.stem
            self.translations[locale] = json.loads(file.read_text())
    
    def get_message(
        self, 
        key: str, 
        locale: str = "en",
        **kwargs
    ) -> str:
        """Get translated message with interpolation."""
        keys = key.split(".")
        value = self.translations.get(locale, {})
        
        for k in keys:
            value = value.get(k, {})
        
        if not isinstance(value, str):
            return key  # Return key if not found
        
        # Interpolation
        for k, v in kwargs.items():
            value = value.replace(f"{{{k}}}", str(v))
        
        return value

i18n = I18nService()

# backend/app/api/deps.py
from fastapi import Request

def get_locale(request: Request) -> str:
    """Extract locale from Accept-Language header."""
    accept_language = request.headers.get("Accept-Language", "en")
    # Parse and return primary locale
    return accept_language.split(",")[0].split("-")[0][:2]
```

### Validation Messages

```python
# backend/app/schemas/user.py
from pydantic import BaseModel, EmailStr, Field, field_validator
from app.core.i18n import i18n

class UserCreate(BaseModel):
    email: EmailStr
    password: str = Field(..., min_length=8)
    
    @field_validator('password')
    @classmethod
    def validate_password(cls, v, info):
        locale = info.context.get('locale', 'en') if info.context else 'en'
        
        if len(v) < 8:
            raise ValueError(
                i18n.get_message('validation.password.minLength', locale, min=8)
            )
        
        if not any(c.isupper() for c in v):
            raise ValueError(
                i18n.get_message('validation.password.uppercase', locale)
            )
        
        return v
```

### Error Responses

```python
# backend/app/api/exceptions.py
from fastapi import Request, HTTPException
from fastapi.responses import JSONResponse
from app.core.i18n import i18n

class LocalizedHTTPException(HTTPException):
    def __init__(
        self, 
        status_code: int, 
        message_key: str,
        **kwargs
    ):
        self.message_key = message_key
        self.message_kwargs = kwargs
        super().__init__(status_code, detail=message_key)

async def localized_exception_handler(
    request: Request, 
    exc: LocalizedHTTPException
):
    locale = get_locale(request)
    
    detail = i18n.get_message(
        exc.message_key, 
        locale, 
        **exc.message_kwargs
    )
    
    return JSONResponse(
        status_code=exc.status_code,
        content={"detail": detail}
    )

# Usage
raise LocalizedHTTPException(
    status_code=404,
    message_key="errors.user.notFound",
    user_id=123
)
```

## RTL (Right-to-Left) Support

```tsx
// frontend/src/app/[locale]/layout.tsx
import { dir } from 'i18n-dir';

const RTL_LOCALES = ['ar', 'he', 'fa', 'ur'];

export default function Layout({ 
  children, 
  params: { locale } 
}: { 
  children: React.ReactNode;
  params: { locale: string };
}) {
  const isRTL = RTL_LOCALES.includes(locale);
  
  return (
    <html lang={locale} dir={isRTL ? 'rtl' : 'ltr'}>
      <body className={isRTL ? 'font-rtl' : 'font-ltr'}>
        {children}
      </body>
    </html>
  );
}

// Tailwind RTL utilities
// tailwind.config.ts
module.exports = {
  theme: {
    extend: {
      // RTL-aware margin/padding
      margin: {
        'start': 'margin-inline-start',
        'end': 'margin-inline-end',
      },
    },
  },
};
```
